#include <include.h>
//#include <SPI/SPIClass.h>
//#include <Serial/Serial.h>
//#include <SD/SD.h>
#include <W2812/W2812.h>

Ws2812 ww[20];

int main()
{
	SysTick_Init();

	ww[0].begin(B12, 3);
	ww[1].begin(B13, 5);

	ww[0].clearAll();
	ww[1].clearAll();


	delay(200);
	while(1)
	{
		ww[0].setAll(0x00, 0x00, 0xFF);
		ww[1].setAll(0xFF, 0xFF, 0x00);
		ww[0].showStrip();
		ww[1].showStrip();
		delay(1000);
		ww[0].setAll(0x00, 0xFF, 0xFF);
		ww[1].setAll(0x00, 0xFF, 0xFF);
		ww[0].showStrip();
		ww[1].showStrip();
		delay(1000);
		ww[0].setAll(0xFF, 0x00, 0xFF);
		ww[1].setAll(0xFF, 0x00, 0xFF);
		ww[0].showStrip();
		ww[1].showStrip();
		delay(1000);
	}
}
